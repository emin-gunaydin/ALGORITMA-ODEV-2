#include <stdio.h>

int main() {
    
    int sayi=0, sayi1=1, sayi2=1, sayi3;
    
    printf("Fibonacci serisinde ilk kaç terimi bulmak istersiniz? : ");
    scanf("%d", &sayi);
    
    printf("Girdiginiz ilk %d terim sayisi asagidaki gibidir;\n", sayi);
    
    for(int i = 1; i<=sayi; i++){
        
        sayi3=sayi1+sayi2;
        sayi1=sayi2;
        sayi2=sayi3;
        
        printf("%d\n", sayi3);
    }
    
    return 0;
}
